Executes a function just once

Example
// sjs.U.generateGUID generates a unique GUID
const uniqueID = sjs.F.once(sjs.U.generateGUID);
// returns a string like "63651124-d561-1897-36c8-53058d87519c"
uniqueID();

// return undefined
uniqueID();

// return undefined
uniqueID();
